	\a{א}\a{ב}\s{ד} \i{לי} \a{ה}\a{כ}\u{דו}\s{ר}\\
	\a{ה}\a{ג}\o{דו}\s{ל} \a{ה}\a{א}\o{דו}\s{ם}.\\
	\o{בו} \i{שי}\a{ח}\s{ק}\i{תי}\\
	\o{כ}\s{ל} \a{ה}\o{יו}\s{ם},

	\u{ו}\a{ב}\a{ל}\s{י}\a{לה}~—\\
	\a{ח}\a{ל}\s{מ}\i{תי}: \e{ת}\o{הו}\s{ם}\\
	\a{ו}\a{א}\i{ני} \o{זו}\e{ר}\s{ק} \a{כ}\u{דו}\s{ר} \a{ל}\a{ש}\a{מ}\i{יי}\s{ם}\\
	\e{ו}\o{שו}\e{ל}\gnuva \s{ח} \a{א}\a{ח}\a{רי}\s{ו} \a{ה}\a{י}\a{ד}\i{יי}\s{ם}\\
	\i{ל}\s{ת}\o{פו}\s{ש}. \a{א}\i{ני} \a{ע}\s{ף}\\
	\u{ו}\i{פ}\s{ת}\o{או}\s{ם}~—\\
	\o{נו}\e{פ}\s{ל} \a{ה}\a{כ}\u{דו}\s{ר} \e{א}\s{ל} \a{ה}\e{ת}\o{הו}\s{ם}\\
	\a{ו}\a{א}\i{ני} \a{א}\a{ח}\a{רי}\s{ו}.

	\a{ב}\o{בו}\e{ק}\s{ר} \i{חי}\a{פ}\s{ש}\i{תי} \a{ל}\a{ש}\s{ו}\s{ו}\s{א}\\
	\a{ב}\i{מי}\a{טה}, \a{ב}\a{א}\o{רו}\s{ן}, \a{ב}\u{שו}\s{ל}\a{ח}\s{ן}\\
	\e{ו}\a{ת}\a{ח}\s{ת} \o{כ}\s{ל} \i{שי}\gnuva \s{ח} \a{ב}\a{ג}\s{ן}~—\\
	\o{לא} \a{ה}\a{יה} \a{ה}\a{כ}\u{דו}\s{ר}.

	\e{ו}\a{א}\s{ז}~—\\
	\a{ע}\a{צ}\s{מ}\i{תי} \a{ה}\e{ע}\s{י}\a{נ}\i{יי}\s{ם}\\
	\e{ו}\i{שי}\a{ח}\s{ק}\i{תי}, \a{כ}\s{ך}, \a{ב}\a{י}\a{ד}\i{יי}\s{ם}:\\
	\a{א}\a{ח}\s{ת}, \s{ש}\a{ת}\i{יי}\s{ם}, \a{ש}\o{לו}\s{ש};\\
	\a{י}\s{ד}, \a{ח}\e{זה} \a{ו}\o{רא}\s{ש}\\
	\i{כ}\s{פ}\a{ל}\i{יי}\s{ם}~—~—~—\\
	\a{ח}\a{ש}\s{ב}\i{תי}, \e{ב}\e{א}\e{מ}\s{ת} \a{ה}\a{כ}\u{דו}\s{ר} \a{ב}\a{י}\a{ד}\i{יי}\s{ם}.

	\a{צ}\a{ח}\a{קה} \a{א}\o{חו}\i{תי} \a{ה}\s{ג}\o{דו}\a{לה}:\\
	„\a{א}\i{חי} \a{ה}\a{ק}\a{ט}\s{ן}, \e{א}\s{י}\s{ך} \i{שי}\a{ח}\s{ק}\a{ת} \a{ה}\o{יו}\s{ם}?”\\
	\e{ו}\a{א}\a{ב}\s{ד} \i{לי} \e{ש}\i{ני}\s{ת} \a{ה}\a{כ}\u{דו}\s{ר} \a{ה}\a{א}\o{דו}\s{ם}.
