{\a{ה}\o{כּו}\a{כ}\x{ב} \a{ה}\a{י}\o{רו}\x{ק}, \e{ס}\e{מ}\x{ל} \a{ה}\e{מ}\u{זו}\e{הה} \i{ע}\x{ם} \a{ה}\e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו}}
\hl{\e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו}} (\L{Esperanto}) \i{היא} \a{שׂ}\a{פה} \e{מ}\u{תו}\x{כ}\e{נ}\e{נ}\x{ת} \e{ש}\o{נו}\x{צ}\a{רה} \e{ב}־1887 \a{ע}\x{ל} \e{י}\e{ד}\x{י} \e{א}\i{לי}\e{ע}\e{ז}\x{ר} (\e{ל}\x{יי}\e{זע}\x{ר}) \u{לו}\x{ד}\i{ווי}\x{ג} (\e{ל}\i{וי}) \a{ז}\e{מ}\x{נ}\o{הו}\x{ף} (\e{ב}\e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו}: \L{Ludoviko Lazaro \textsc{Zamenhofo}} [\u{לו}\x{ד}\i{ווי}\o{קו} \a{ל}\a{ז}\o{רו} \a{ז}\e{מ}\x{נ}\o{הו}\o{פו}]; 15 \e{ב}\e{ד}\e{צ}\x{מ}\e{בּ}\x{ר} 1859~— 14 \e{ב}\a{א}\x{פּ}\i{רי}\x{ל} 1917), \e{ש}\i{יי}\e{ע}\x{ד} \o{או}\a{תה} \a{ל}\a{ה}\o{פו}\x{ך} \i{ל}\x{שׂ}\a{פ}\o{תו} \a{ה}\x{ש}\i{ני}\a{יה} \e{ש}\x{ל} \o{כ}\x{ל} \a{א}\a{ד}\x{ם} \a{ב}\o{עו}\a{ל}\x{ם}, \e{ל}\a{צ}\x{ד} \x{שׂ}\a{פ}\x{ת} \a{ה}\e{א}\x{ם} \e{ש}\o{לו}, \e{ב}\a{מ}\a{ט}\a{רה} \e{ל}\a{ק}\e{ר}\x{ב} \e{ל}\a{ב}\o{בו}\x{ת} \e{ב}\x{י}\x{ן} \x{ב}\e{נ}\x{י}־\a{א}\a{ד}\x{ם} \o{דו}\x{ב}\e{ר}\x{י} \a{שׂ}\o{פו}\x{ת} \o{שו}\o{נו}\x{ת}, \e{ב}\e{א}\x{מ}\a{צ}\u{עו}\x{ת} \a{שׂ}\a{פה} \e{נ}\x{יי}\x{ט}\a{ר}\i{לי}\x{ת}. \e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו} \i{היא} \a{כ}\o{יו}\x{ם} \a{ה}\a{שׂ}\a{פה} \a{ה}\e{מ}\u{דו}\e{בּ}\e{ר}\x{ת} \e{ב}\o{יו}\e{ת}\x{ר} \i{מ}\e{בּ}\x{י}\x{ן} \a{ה}\a{שׂ}\o{פו}\x{ת} \a{ה}\e{מ}\u{תו}\x{כ}\a{נ}\o{נו}\x{ת}.

\e{ב}־1887, \e{ל}\a{א}\a{ח}\x{ר} \e{ל}\a{מע}\a{לה} \e{מ}\e{ע}\e{שׂ}\x{ר} \x{ש}\o{נו}\x{ת} \a{ע}\o{בו}\a{דה}, \a{י}\a{צ}\x{ר} \a{ז}\e{מ}\x{נ}\o{הו}\x{ף}, \e{ש}\a{ה}\a{יה} \o{רו}\e{פא}, \e{א}\x{ת} \a{ה}\e{י}\o{סו}\x{ד} \a{ל}\a{שׂ}\a{פה}. \u{הוא} \i{ה}\x{ת}\a{כּ}\e{וו}\x{ן} \i{לי}\o{צו}\x{ר} \a{שׂ}\a{פה} \a{ק}\a{לה} \e{ל}\i{לי}\u{מו}\x{ד} \e{ש}\e{ת}\a{ש}\e{מ}\x{ש} \x{שׂ}\a{פ}\x{ת}־\e{ע}\e{ז}\x{ר} \x{כ}\a{ל}\x{ל}־\o{עו}\a{ל}\i{מי}\x{ת}, \a{שׂ}\a{פה} \x{ש}\i{ני}\a{יה} \e{ל}\o{כ}\x{ל} \a{א}\a{ד}\x{ם} \a{ב}\o{עו}\a{ל}\x{ם}. \a{כ}\o{יו}\x{ם}, \a{א}\x{ל}\e{פ}\x{י} \a{א}\a{נ}\i{שי}\x{ם} \i{מ}\x{ש}\a{ת}\x{מ}\i{שי}\x{ם} \e{ב}\e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו} \e{ב}\o{או}\e{פ}\x{ן} \a{ק}\u{בו}\a{ע} \e{כ}\e{ד}\x{י} \e{ל}\a{ת}\x{ק}\e{ש}\x{ר} \i{ע}\x{ם} \a{א}\a{נ}\i{שי}\x{ם} \i{מ}\o{כּ}\x{ל} \a{ה}\o{עו}\a{ל}\x{ם}. \a{ע}\x{ל} \a{ה}\o{חו}\e{ב}\e{ר}\x{ת} \e{ש}\x{ש}\a{מה} „\a{ה}\a{שׂ}\a{פה} \a{ה}\e{בּ}\x{י}\x{נ}\e{ל}\u{או}\i{מי}\x{ת}~— \a{מ}\o{בוא} \e{ו}\e{ס}\e{פ}\x{ר}־\i{לי}\u{מו}\x{ד} \a{מ}\e{לא}” \a{ח}\a{ת}\x{ם} ד״ר \a{ז}\e{מ}\x{נ}\o{הו}\x{ף}, \a{כ}\e{מ}\u{קו}\a{בּ}\x{ל} \e{ב}\o{או}\a{ת}\x{ם} \a{י}\i{מי}\x{ם}, \e{ב}\e{ש}\x{ם}־\e{ע}\x{ט}, „\L{Doktoro Esperanto}” [\o{דו}\x{ק}\o{טו}\o{רו} \e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו}], \e{ש}\e{פּי}\u{רו}\o{שו} „\a{ה}\o{דו}\x{ק}\o{טו}\x{ר} \a{ה}\e{מ}\a{ק}\e{ווה}”, \e{ש}\e{כּ}\x{ן} \a{ז}\e{מ}\x{נ}\o{הו}\x{ף} \i{קי}\a{ווה} \e{ש}\e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו} \a{ת}\a{ה}\o{פו}\x{ך} \i{ל}\x{שׂ}\a{פ}\x{ת}־\e{ע}\e{ז}\x{ר} \e{ב}\x{י}\x{נ}\e{ל}\u{או}\i{מי}\x{ת} \e{ו}\i{ת}\x{ת}\o{רו}\x{ם} \a{ל}\a{ש}\o{לו}\x{ם} \a{ב}\o{עו}\a{ל}\x{ם}.

\a{ה}\i{ד}\x{ק}\u{דו}\x{ק} \e{ש}\x{ל} \a{ה}\e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו} \a{פ}\u{שו}\x{ט} \a{י}\a{ח}\i{סי}\x{ת}. \e{י}\x{ש} \a{בה} 16 \x{כ}\a{ל}\e{ל}\x{י} \i{ד}\x{ק}\u{דו}\x{ק} \x{ב}\i{סי}\i{סי}\i{י}\x{ם}. \o{או}\a{צ}\x{ר}־\a{ה}\i{מי}\i{לי}\x{ם} \a{ה}\x{בּ}\i{סי}\i{סי} \a{ב}\u{נו}\x{י} \a{ע}\x{ל} \e{י}\o{סו}\x{ד} \i{מ}\x{ס}\a{פּ}\x{ר} \e{מ}\u{צו}\x{מ}\a{צ}\x{ם} \a{י}\a{ח}\i{סי}\x{ת} \e{ש}\x{ל} \o{שו}\a{ר}\i{שי}\x{ם}, \e{ו}\e{א}\x{י}\x{ן} \a{בה} \o{יו}\x{צ}\i{אי}\x{ם} \i{מ}\x{ן} \a{ה}\x{כּ}\a{ל}\x{ל}. \i{מ}\u{שו}\x{ם} \a{כ}\x{ך}, \e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו} \a{ק}\a{לה} \o{יו}\e{ת}\x{ר} \e{ל}\i{לי}\u{מו}\x{ד} \i{מ}\a{שׂ}\a{פה} \a{ז}\a{רה} \a{א}\e{ח}\e{ר}\x{ת}, \i{ב}\x{מ}\u{יו}\a{ח}\x{ד} \e{ל}\o{דו}\x{ב}\e{ר}\x{י} \a{שׂ}\o{פו}\x{ת} \e{א}\x{י}\o{רו}\i{פּ}\o{יו}\x{ת}. \e{ב}\e{ד}\e{ר}\x{ך}־\x{כ}\a{ל}\x{ל}, \a{כ}\a{מה} \o{חו}\a{ד}\i{שי}\x{ם} \e{ש}\x{ל} \i{לי}\u{מו}\x{ד} \e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו} \x{ש}\u{קו}\i{לי}\x{ם} \e{ל}\i{מ}\x{ס}\a{פּ}\x{ר} \x{ש}\o{נו}\x{ת} \i{לי}\u{מו}\x{ד} \e{ב}\a{שׂ}\o{פו}\x{ת} \a{א}\e{ח}\o{רו}\x{ת}, \u{דו}\x{ג}\a{מ}\x{ת} \a{א}\x{נ}\x{ג}\i{לי}\x{ת} \o{או} \a{צ}\x{ר}\a{פ}\i{תי}\x{ת}, \a{ו}\a{א}\i{פי}\u{לו} \e{ל}\o{דו}\x{ב}\e{ר}\x{י} \a{שׂ}\o{פו}\x{ת} \e{ש}\e{א}\x{י}\a{נ}\x{ן} \e{א}\x{י}\o{רו}\i{פּ}\o{יו}\x{ת} \a{ק}\x{ל} \o{יו}\e{ת}\x{ר} \i{ל}\x{ל}\o{מו}\x{ד} \e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו} \e{מ}\a{א}\e{ש}\x{ר} \o{כ}\x{ל} \a{שׂ}\a{פה} \e{א}\x{י}\o{רו}\i{פּי}\x{ת} \a{א}\e{ח}\e{ר}\x{ת}.

\e{י}\x{ש} \a{כ}\a{מה} \e{מ}\o{או}\x{ת} \a{ע}\x{ד} \a{א}\x{ל}\e{פ}\x{י} \o{דו}\x{ב}\e{ר}\x{י} \e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו} \i{מ}\e{ל}\x{י}\a{דה} (\a{ה}\u{מו}\a{נ}\x{ח} \a{ה}\e{מ}\u{קו}\a{בּ}\x{ל} \u{הוא} „\L{denaskaj esperantistoj}” [\e{ד}\a{נ}\x{ס}\a{ק}\x{י} \e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\i{טי}\x{ס}\o{טו}\x{י}]), \e{ו}\a{ק}\a{יי}\o{מו}\x{ת} \i{מ}\x{ש}\a{פּ}\o{חו}\x{ת} \a{ה}\i{מ}\x{ש}\a{ת}\x{מ}\o{שו}\x{ת} \a{ב}\a{שׂ}\a{פה} \i{כ}\x{שׂ}\a{פ}\x{ת} \a{ה}\i{די}\u{בּו}\x{ר} \a{ב}\a{בּ}\i{י}\x{ת}. \i{ב}\x{ז}\u{כו}\x{ת} \i{ק}\u{יו}\a{מ}\x{ם} \e{ש}\x{ל} \o{דו}\x{ב}\i{רי}\x{ם} \e{י}\i{לי}\i{די}\i{י}\x{ם} \a{ל}\a{שׂ}\a{פה}, \i{ני}\a{ת}\x{ן} \e{ל}\a{ה}\x{ג}\i{די}\x{ר} \o{או}\a{תה} \a{כ}\o{יו}\x{ם} \e{כ}„\a{שׂ}\a{פה} \i{ט}\x{ב}\i{עי}\x{ת}”, \a{ג}\x{ם} \i{א}\x{ם} \e{מ}\o{קו}\a{רה} \e{מ}\u{תו}\x{כ}\a{נ}\x{ן} \e{ב}\u{מו}\a{דע}, \a{מ}\a{ע}\e{שׂה} \e{י}\e{ד}\x{י} \a{א}\a{ד}\x{ם}.

\a{ק}\x{ל} \e{מ}\o{או}\x{ד} \i{ל}\x{ל}\o{מו}\x{ד} \e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו}: \a{ג}\x{ם} \a{ה}\x{כּ}\a{ת}\x{ב}, \a{ג}\x{ם} \a{ה}\i{ד}\x{ק}\u{דו}\x{ק} \e{ו}\a{ג}\x{ם} \o{או}\a{צ}\x{ר}־\a{ה}\i{מי}\i{לי}\x{ם}. \i{א}\x{ם} \a{א}\e{ת}\x{ם} \x{כ}\a{ב}\x{ר} \o{יו}\x{ד}\i{עי}\x{ם} \a{שׂ}\a{פה} \e{א}\x{י}\o{רו}\i{פּי}\x{ת}, \i{ב}\x{מ}\u{יו}\a{ח}\x{ד} \o{רו}\a{מא}\i{ני}\x{ת}\footnote{\i{אי}\a{ט}\x{ל}\i{קי}\x{ת}, \x{ס}\a{פ}\a{ר}\i{די}\x{ת}, \a{צ}\x{ר}\a{פ}\i{תי}\x{ת}, \a{ק}\a{ט}\a{ל}\i{ני}\x{ת}, \o{פו}\x{ר}\u{טו}\e{ג}\i{זי}\x{ת}, \a{לא}\i{די}\o{נו}, \o{רו}\a{מ}\i{ני}\x{ת} \a{ו}\a{א}\e{ח}\o{רו}\x{ת}, \o{או} \i{א}\a{מא} \e{ש}\a{ל}\e{ה}\x{ן}, \a{ה}\a{ל}\i{טי}\i{ני}\x{ת}.}, \i{י}\x{ה}\e{יה} \a{ל}\e{כ}\x{ם} \a{ק}\x{ל} \i{ב}\x{מ}\u{יו}\a{ח}\x{ד} \i{ל}\x{ל}\o{מו}\x{ד} \e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו}; \i{א}\x{ם} \a{א}\e{ת}\x{ם} \o{לא} \o{יו}\x{ד}\i{עי}\x{ם}, \i{לי}\u{מו}\x{ד} \a{ה}\a{שׂ}\a{פה} \i{י}\e{ת}\x{ן} \a{ל}\e{כ}\x{ם} \a{ב}\i{סי}\x{ס} \o{טו}\x{ב} \o{לא} \a{ר}\x{ק} \x{כ}\e{ש}\i{ת}\x{ר}\u{צו} \i{ל}\x{ל}\o{מו}\x{ד} \a{שׂ}\a{פה} \e{א}\x{י}\o{רו}\i{פּי}\x{ת}~— \u{ו}\i{ב}\x{פ}\a{ר}\x{ט}, \a{כ}\a{א}\u{מו}\x{ר}, \o{רו}\a{מא}\i{ני}\x{ת}~— \e{א}\a{לא} \i{ב}\x{כ}\a{ל}\x{ל} \e{ב}\i{לי}\u{מו}\x{ד} \a{שׂ}\a{פה} \a{ז}\a{רה}. \a{ק}\a{יי}\i{מי}\x{ם} \x{ס}\a{פ}\i{רי}\x{ם} \a{ר}\i{בּי}\x{ם} \e{ש}\u{תו}\x{כ}\u{לו} \i{ל}\x{ק}\o{רוא} \e{ב}\e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו}, \e{ה}\x{ן} \i{ס}\x{פ}\u{רו}\x{ת} \a{מ}\o{קו}\x{ר} \e{ו}\e{ה}\x{ן} \e{מ}\u{תו}\x{ר}\e{ג}\e{מ}\x{ת}, \e{ו}\u{תו}\x{כ}\u{לו} \e{ל}\i{ה}\a{פּ}\e{ג}\x{ש}, \e{ל}\a{ד}\e{בּ}\x{ר} \u{ו}\x{ל}\i{ה}\x{ת}\a{כּ}\e{ת}\x{ב} \i{ע}\x{ם} \a{א}\a{נ}\i{שי}\x{ם} \i{מ}\o{כּ}\x{ל} \a{ר}\a{ח}\e{ב}\x{י} \a{ה}\o{עו}\a{ל}\x{ם} \a{ע}\x{ל} \a{ב}\i{סי}\x{ס} \i{ש}\x{וו}\o{יו}\i{ני}.

\hl{\o{הו}\a{מ}\a{ר}\i{ני}\x{ס}\o{מו}} (\e{ב}\e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו}: \L{Homaranismo}; „\e{א}\o{נו}\i{ש}\u{יו}\x{ת}”, „\a{ה}\a{ה}\x{ש}\a{ק}\a{פה} \e{ש}\x{ל} \e{ה}\o{יו}\x{ת} \e{ח}\e{ל}\x{ק} \i{מ}\i{מ}\x{ש}\a{פּ}\a{ח}\x{ת} \a{ה}\a{א}\a{ד}\x{ם}”), \i{היא} \i{מי}\a{לה} \e{ב}\e{א}\x{ס}\e{פּ}\a{ר}\x{נ}\o{טו} \e{ש}\a{בּה} \i{ה}\x{ש}\a{ת}\e{מ}\x{ש} \a{ז}\e{מ}\x{נ}\o{הו}\x{ף} \e{כ}\e{ד}\x{י} \e{ל}\a{ת}\e{א}\x{ר} \e{א}\x{ת} \a{ה}\x{ש}\a{ק}\a{פ}\x{ת} \o{עו}\a{ל}\o{מו} \a{ע}\x{ל} \a{ה}\i{ה}\x{ת}\a{נ}\a{ה}\u{גו}\x{ת} \e{ו}\a{י}\a{ח}\e{ס}\x{י} \a{ה}\o{גו}\x{מ}\i{לי}\x{ן} \e{ב}\x{י}\x{ן} \a{א}\a{ד}\x{ם} \e{ל}\a{א}\a{ד}\x{ם}.
