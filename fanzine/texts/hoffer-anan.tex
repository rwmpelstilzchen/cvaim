\textsection{\e{א}\x{י}\o{פה} \o{כו}\e{א}\x{ב}?}

\begin{twocol}
\begin{stanza}
\u{שו}\x{ב} \a{ב}\o{או}\x{ת} \a{ה}\x{ד}\a{מ}\o{עו}\x{ת} \e{ו}\u{כו}\a{ל}\x{ם} \o{שו}\a{א}\i{לי}\x{ם}:\\

\begin{stanza}
\u{שו}\x{ב} \a{ע}\a{מ}\i{לי} \a{ש}\e{כ}\a{חה}\\

\a{ה}\e{ר}\x{י} \o{לא} \a{ת}\i{מי}\x{ד} \a{ע}\a{מ}\i{לי} \o{בו}\a{כה}\\
\a{ר}\x{ק} \i{ב}\x{ג}\a{ל}\x{ל} \e{ש}\i{היא} \i{קי}\x{בּ}\a{לה} \a{מ}\a{כּה}\\
\e{ו}\o{לא} \a{ת}\i{מי}\x{ד} \e{א}\x{פ}\a{ש}\x{ר} \e{ל}\a{ה}\x{ר}\o{או}\x{ת}\\
\i{סי}\a{מ}\x{ן} \o{או} \e{פ}\a{צע} \e{ב}\o{תו}\x{ר} \o{הו}\a{כ}\a{חה}.
