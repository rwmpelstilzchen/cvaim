\textsection{\a{ה}\o{כּ}\x{ל} \a{ק}\a{רה} \i{ב}\x{ג}\a{ל}\x{ל} \u{רו}\i{תי}}

\begin{twocol}
\begin{stanza}
\a{ה}\o{כּ}\x{ל} \a{ק}\a{רה} \i{ב}\x{ג}\a{ל}\x{ל} \u{רו}\i{תי}\\
\a{ה}\a{ל}\x{כ}\u{נו} \i{ל}\x{ק}\o{טו}\x{ף} \a{ל}\u{נו} \u{תו}\i{תי}\x{ם}.\\
\a{ק}\a{ט}\x{פ}\u{נו} \a{ק}\a{ט}\x{פ}\u{נו}\\
\a{ע}\x{ד} \e{ש}\a{ע}\a{יי}\x{פ}\u{נו}\\
\e{ו}\a{ע}\x{ד} \e{ש}\a{ה}\a{יה} \e{מ}\u{או}\a{ח}\x{ר}.\\
\x{ק}\a{צ}\x{ת} \a{א}\a{כ}\x{ל}\u{נו},\\
\x{ק}\a{צ}\x{ת} \i{ק}\x{ל}\a{ק}\x{ל}\u{נו},\\
\x{ק}\a{צ}\x{ת} \a{ג}\x{ם} \i{נ}\x{ש}\a{א}\x{ר} \e{ל}\a{מ}\a{ח}\x{ר}.
\end{stanza}

\begin{stanza}
\i{טי}\a{פּ}\x{ס}\u{נו} \a{ע}\x{ל} \o{כ}\x{ל} \a{ה}\e{ע}\i{צי}\x{ם},\\
\a{ה}\u{תו}\i{תי}\x{ם} \a{ה}\u{יו} \a{ח}\u{מו}\i{צי}\x{ם}.\\
\i{חי}\a{ל}\x{ק}\u{נו} \e{ל}\o{כ}\x{ל} \a{ה}\e{י}\a{ל}\i{די}\x{ם}\\
\i{ה}\x{ש}\i{חי}\u{רו} \a{ל}\e{ה}\x{ם} \a{ה}\x{בּ}\a{ג}\i{די}\x{ם}.\\
\a{א}\i{פי}\u{לו} \a{א}\i{מי}\x{ר} \a{ה}\a{ק}\a{ט}\x{ן}\\
\a{צ}\x{מ}\u{חו} \o{לו} \a{שׂ}\a{פ}\x{ם} \e{ו}\a{ז}\a{ק}\x{ן}.
\end{stanza}

\begin{stanza}
\e{ל}\u{עו}\i{זי} \i{ה}\x{ש}\i{חי}\u{רו} \o{א}\x{ז}\a{ני}\x{ו},\\
\e{ל}\u{זו}\i{זי} \i{ה}\x{ש}\i{חי}\x{ר} \a{ה}\a{ז}\a{נ}\x{ב}.\\

\end{twocol}



\textsection{\x{ז}\a{ר}\i{עי}\x{ם} \e{ש}\x{ל} \a{מ}\x{ס}\i{טי}\x{ק}}

\begin{twocol}
\begin{stanza}
\e{ל}\e{נ}\a{טע} \e{ו}\i{לי} \e{י}\x{ש} \o{סו}\x{ד} \e{נ}\x{ח}\a{מ}\x{ד}\\
\e{ש}\o{עו}\x{ד} \o{לא} \i{סי}\a{פּ}\x{ר}\u{נו} \e{ל}\a{א}\x{ף} \e{א}\a{ח}\x{ד}.\\
\a{א}\a{נ}\x{ח}\u{נו} \a{ז}\a{רע}\u{נו} \i{מ}\a{ת}\a{ח}\x{ת} \a{ל}\x{פּ}\a{ל}\x{ס}\i{טי}\x{ק}\\
\x{ז}\a{ר}\i{עי}\x{ם} \e{ש}\x{ל} \a{מ}\x{ס}\i{טי}\x{ק}, \x{ז}\a{ר}\i{עי}\x{ם} \e{ש}\x{ל} \a{מ}\x{ס}\i{טי}\x{ק}.
\end{stanza}

\begin{stanza}
\a{ב}\a{ק}\i{י}\x{ץ} \a{ה}\a{בּא}, \a{ב}\a{ק}\i{י}\x{ץ} \a{ה}\a{בּא}\\
\a{נ}\u{רו}\x{ץ} \a{ל}\i{גי}\a{נה} \e{ב}\o{כ}\x{ל} \o{יו}\x{ם},\\
\e{ו}\a{ע}\x{ל} \o{כ}\x{ל} \i{ג}\x{ב}\o{עו}\x{ל}, \e{ו}\a{ע}\x{ל} \o{כ}\x{ל} \a{ע}\a{נ}\x{ף}\\
\a{י}\x{ב}\i{שי}\x{ל} \a{ל}\u{נו} \a{מ}\x{ס}\i{טי}\x{ק} \a{א}\o{דו}\x{ם}.
\end{stanza}

\begin{stanza}
\a{ב}\a{ח}\x{ר}\u{נו} \a{ב}\a{מ}\x{ס}\i{טי}\x{ק} \a{ה}\i{כי} \a{ט}\i{עי}\x{ם},\\
\a{ח}\a{ת}\x{כ}\u{נו} \o{או}\o{תו} \i{ל}\x{ש}\o{מו}\a{נה} \x{ז}\a{ר}\i{עי}\x{ם}\\
\a{ע}\a{ד}\x{ר}\u{נו}, \a{מ}\a{ד}\x{ד}\u{נו}, \i{סי}\a{מ}\u{נו} \e{ב}\e{ס}\e{ר}\x{ט},\\
\e{ה}\a{כ}\u{נו} \a{ל}\a{מ}\x{ס}\i{טי}\x{ק} \i{גי}\a{נה} \e{נ}\e{ה}\e{ד}\e{ר}\x{ת}.
\end{stanza}

\begin{stanza}
\i{כ}\x{מ}\a{ע}\x{ט} \e{ב}\o{כ}\x{ל} \e{ע}\e{ר}\x{ב} \a{א}\a{נ}\x{ח}\u{נו} \a{ב}\o{או}\x{ת},\\
\a{צ}\i{רי}\x{ך} \e{ל}\a{נ}\e{כּ}\x{ש} \e{ו}\a{צ}\i{רי}\x{ך} \e{ל}\a{ה}\x{ש}\o{קו}\x{ת}.\\
\a{א}\a{נ}\x{ח}\u{נו} \o{עו}\x{ב}\o{דו}\x{ת} \o{פה} \e{ב}\o{או}\e{פ}\x{ן} \a{ק}\u{בו}\a{ע}\\
\a{ש}\o{לו}\x{ש} \e{פ}\a{ע}\i{מי}\x{ם} \e{ו}\o{יו}\e{ת}\x{ר} \e{ב}\a{ש}\u{בו}\a{ע}.
\end{stanza}

\begin{stanza}
\e{ב}\x{י}\x{נ}\a{ת}\i{יי}\x{ם} \a{ה}\a{מ}\x{ס}\i{טי}\x{ק} \o{עו}\x{ד} \o{לא} \a{נ}\a{ב}\x{ט}\\
\a{א}\x{ך} \e{נ}\a{טע} \o{או}\e{מ}\e{ר}\x{ת} \e{ש}\o{לא} \i{א}\x{כ}\a{פּ}\x{ת}.\\
\a{ה}\a{מ}\x{ס}\i{טי}\x{ק} \i{י}\x{נ}\o{בו}\x{ט}, \a{כ}\a{כה} \e{נ}\a{טע} \o{או}\e{מ}\e{ר}\x{ת},\\
\a{פ}\u{שו}\x{ט}, \i{כי} \x{כ}\a{ב}\x{ר} \e{א}\x{י}\x{ן} \o{לו} \x{ב}\e{ר}\x{י}\a{רה} \a{א}\e{ח}\e{ר}\x{ת}.
\end{stanza}
\end{twocol}
