\u{ו}\a{מה} \a{ש}\x{ם} \a{ר}\i{אי}\u{נו}, \u{ו}\a{מה} \a{ש}\x{ם} \i{די}\a{ב}\x{ר}\u{נו},\\
\e{ו}\e{א}\x{י}\x{ך} \a{ש}\x{ם} \i{חי}\a{דה} \e{מ}\u{צו}\e{יי}\e{נ}\x{ת} \a{פ}\a{ת}\x{ר}\u{נו}.

\a{ע}\x{מ}\u{דו} \a{ש}\x{ם} \e{ע}\i{צי}\x{ם} \i{מ}\a{ס}\i{בי}\x{ב} \e{כ}\i{מ}\x{ס}\e{ג}\e{ר}\x{ת},\\
\a{ע}\x{מ}\u{דו} \e{ו}\i{ה}\i{קי}\u{פו} \x{ב}\e{ר}\x{י}\a{כה} \e{מ}\o{פו}\e{א}\e{ר}\x{ת}:\\
\o{כ}\x{ל} \e{ע}\x{ץ} \a{ב}\a{ע}\x{ל}־\u{גו}\x{ף} \e{ו}\a{ז}\u{קו}\x{ף} \e{כ}\a{א}\a{נ}\x{ך},\\
\e{ו}\i{היא} \x{כ}\o{מו} \e{ר}\i{אי} \a{ע}\x{ל} \a{ה}\a{א}\e{ר}\x{ץ} \u{מו}\a{נ}\x{ח}.

\a{ע}\x{מ}\a{דה} \e{ו}\i{ה}\i{בי}\a{טה} \a{ב}\x{ר}\i{אי} \a{ח}\u{מו}\a{ט}\i{לי},\\
\i{ה}\i{בי}\a{טה}, \i{ה}\i{בי}\a{טה}~— \e{ו}\a{כ}\a{כה} \a{א}\x{מ}\a{רה} \i{לי}:

„\e{ר}\e{אה}, \a{א}\x{ב}\a{ר}\a{ה}\x{ם}, \a{ה}\a{מ}\x{ר}\e{אה} \e{ש}\a{ל}\a{מ}\i{י}\x{ם},\\
\e{ר}\e{אה}, \e{א}\x{י}\x{ך} \a{צ}\i{פי}\x{ם} \e{ב}\o{תו}\a{כה} \a{ה}\a{ש}\a{מ}\i{יי}\x{ם},\\
\e{ר}\e{אה}, \e{א}\x{י}\x{ך} \a{ה}\e{ע}\x{ץ} \a{ש}\x{ם} \a{ג}\e{ד}\x{ל} \e{ב}\i{הי}\u{פו}\x{ך},\\
\e{ר}\e{אה}, \e{א}\x{י}\x{ך} \a{ה}\o{או}\x{ר} \a{ש}\x{ם} \o{זו}\e{ר}\a{ע} \i{חי}\u{ו}\x{ך}.

\e{א}\x{י}\x{ך} \a{ע}\x{ב} \e{ב}\o{ח}\x{ל}\a{פה} \a{ה}\a{צ}\o{מו}\x{ת} \a{בה} \o{סו}\e{ר}\e{ק}\x{ת},\\
\e{ו}\e{ע}\x{י}\x{ן} \a{ה}\x{ב}\e{ר}\x{י}\a{כה} \o{צו}\i{פ}\a{יה} \e{ו}\o{צו}\e{ח}\e{ק}\x{ת}.”

\o{רו}\e{ב}\e{צ}\x{ת} \x{ב}\e{ר}\x{י}\a{כה} \o{צו}\a{ח}\a{קה} \u{ו}\x{ד}\u{מו}\a{מ}ה,\\
\o{צו}\e{ח}\e{ק}\x{ת} \a{ג}\x{ם} \a{ט}\i{לי} \e{ב}\o{או}\x{ר} \a{ה}\a{ח}\a{מה}.\\
\a{א}\a{מ}\x{ר}\i{תי} \e{ל}\a{ט}\i{לי}:

„\a{ה}\i{גי}\i{די} \i{לי}, \a{ט}\i{לי},\\
\a{מ}\u{דו}\a{ע} \i{פ}\x{ת}\o{או}\x{ם} \a{ה}\x{ב}\e{ר}\x{י}\a{כה} \o{צו}\a{ח}\a{קה} \i{לי}?\\
\e{ו}\a{ל}\a{מה} \e{מ}\x{י}\e{מי}\a{ה} \a{ז}\i{כי}\x{ם} \u{ו}\x{ש}\e{ק}\i{טי}\x{ם},\\
\u{ו}\x{ל}\e{פ}\a{תע} \i{נ}\x{צ}\u{נו}\x{ץ} \i{ב}\x{פ}\e{נ}\x{י}\e{ה}\x{ם} \u{ו}\x{ק}\a{מ}\i{טי}\x{ם}?”

\a{ש}\x{מ}\a{עה} \a{ח}\u{מו}\a{ט}\i{לי},\\
\a{ש}\x{מ}\a{עה} \e{ו}\a{נ}\o{בו}\a{כה},\\
\a{ש}\x{מ}\a{עה} \e{ו}\a{א}\x{מ}\a{רה} \i{לי}:\\
„\a{ח}\a{כ}\x{ם} \e{ש}\a{כ}\o{מו}\a{ך}!\\
\a{א}\a{תה} \a{ה}\x{מ}\a{ת}\e{ר}\x{ץ} \i{לי} \a{ת}\i{מי}\x{ד} \e{א}\x{ת} \a{ה}\o{כ}\x{ל},\\
\a{מה} \e{זה} \i{פ}\x{ת}\o{או}\x{ם} \e{ש}\i{ה}\x{ת}\a{ח}\x{ל}\a{ת} \i{ל}\x{ש}\o{או}\x{ל}?”

„\a{ח}\x{ד}־\e{א}\e{פ}\x{ס}!”~— \a{א}\a{מ}\x{ר}\i{תי} \e{ל}\a{ט}\i{לי} (\i{פי}\x{ק}\i{חי}\x{ת} \i{היא}!)\\
\a{ח}\a{ש}\x{ב}\i{תי}, \a{ח}\a{ש}\x{ב}\i{תי}~— \e{ו}\a{כ}\a{כה} \a{ע}\i{ני}\i{תי}:

„\a{ה}\e{א}\x{י}\x{ן} \e{זה} \i{מ}\x{פ}\e{נ}\x{י} \e{ש}\a{ד}\i{גי}\x{ם} \u{ו}\x{ד}\i{גי}\i{גי}\x{ם}\\
\o{חו}\x{ל}\i{פי}\x{ם} \e{ב}\o{מו}\a{חה} \e{כ}\a{ח}\o{לו}\x{ף} \a{ה}\i{גי}\i{גי}\x{ם},\\
\e{ו}\e{ה}\x{ם} \i{ב}\x{ק}\a{מ}\i{טי}\x{ם} \a{ע}\x{ל} \i{מ}\x{צ}\a{חה} \e{י}\a{ק}\x{מ}\u{טו}\a{ה},~—\\
\i{כי} \a{כ}\a{כה} \o{חו}\x{ש}\i{בי}\x{ם} \a{מ}\x{ח}\a{ש}\o{בו}\x{ת}, \a{כ}\a{י}\u{דו}\a{ע}.”

\a{ע}\x{מ}\a{דה} \e{ו}\i{ה}\x{ק}\i{שי}\a{בה} \u{דו}\a{מ}\x{ם} \a{ח}\u{מו}\a{ט}\i{לי},\\
\i{ה}\x{ק}\i{שי}\a{בה}, \i{ה}\x{ק}\i{שי}\a{בה}~— \e{ו}\a{כ}\a{כה} \a{ע}\x{נ}\a{תה} \i{לי}:

„\i{א}\x{ם} \a{כ}\a{כה}, \a{א}\x{ז} \a{ל}\a{מה} \u{כו}\a{ל}\x{ם} \o{נו}\a{ה}\i{גי}\x{ם}\\
\i{ל}\x{ק}\o{רוא} \i{ל}\x{ב}\i{ר}\o{יו}\x{ת} \e{ש}\a{כ}\e{א}\e{לה} \a{ד}\i{גי}\x{ם}?”

\a{מ}\a{ש}\x{כ}\i{תי} \a{כ}\e{ת}\x{ף} \e{ו}\a{א}\a{מ}\x{ר}\i{תי} \e{ל}\a{ט}\i{לי}:\\
„\i{מ}\x{נ}\a{ה}\x{ג} \e{זה} \a{פ}\u{שו}\x{ט}, \e{ו}\e{א}\x{י}\o{נו} \e{ל}\i{חי}\a{דה} \i{לי}.\\
\a{מה} \e{י}\x{ש} \a{כא}\x{ן} \i{ל}\x{ת}\o{מו}\a{ה} \a{ע}\x{ל} \o{סו}\x{ד} \a{ה}\i{מ}\x{נ}\a{ה}\x{ג}?\\
\a{ה}\a{ד}\x{ג} \e{ז}\u{הו} \a{ד}\x{ג},~—\\
\e{ו}\a{ע}\x{ל}־\e{כ}\x{ן} \x{ש}\o{מו} \o{הוא} \a{ד}\x{ג}!”

\a{ע}\x{מ}\a{דה} \e{ו}\i{ה}\x{ת}\i{מי}\a{הה} \a{ע}\a{ל}\x{י}, \a{ח}\u{מו}\a{ט}\i{לי},\\
\i{ה}\x{ת}\i{מי}\a{הה}, \i{ה}\x{ת}\i{מי}\a{הה}~— \e{ו}\a{כ}\a{כה} \a{ע}\x{נ}\a{תה} \i{לי}:

„\o{לא}, \e{א}\x{י}\x{ן} \e{זה} \a{פ}\u{שו}\x{ט}, \e{י}\i{די}\i{די} \a{א}\x{ב}\a{ר}\a{ה}\x{ם}!\\
\e{א}\x{י}\x{ן} \e{ש}\x{ם} \a{ב}\o{עו}\a{ל}\x{ם} \a{ה}\i{ני}\a{ת}\x{ן} \e{ל}\i{חי}\a{נ}\x{ם},\\
\e{ל}\o{כ}\x{ל} \a{ה}\e{ש}\o{מו}\x{ת} \e{י}\x{ש} \i{סי}\a{בה} \e{ו}\e{י}\x{ש} \a{ט}\a{ע}\x{ם}.”

\a{א}\a{מ}\x{ר}\i{תי}:\\
„\i{א}\x{ם} \a{כ}\x{ך}, \a{א}\x{ת} \a{ה}\x{ס}\i{בי}\i{רי} \a{ה}\a{פ}\a{ע}\x{ם}.\\
\a{ת}\i{מי}\x{ד} \a{א}\x{ת} \o{שו}\e{א}\e{ל}\x{ת}~— \a{א}\i{ני} \a{ה}\e{מ}\i{שי}\x{ב},\\
\a{ה}\o{יו}\x{ם} \a{א}\x{ת} \o{עו}\a{נה}~— \a{ו}\a{א}\i{ני} \a{ה}\a{מ}\x{ק}\i{שי}\x{ב}.”

\a{ה}\x{ר}\e{בה} \a{מ}\x{ח}\a{ש}\o{בו}\x{ת} \a{א}\x{ז} \a{ח}\x{ש}\a{בה} \a{ח}\u{מו}\a{ט}\i{לי},\\
\a{ח}\x{ש}\a{בה} \e{ו}\a{ח}\x{ש}\a{בה} \a{לה}~— \e{ו}\a{כ}\a{כה} \a{ע}\x{נ}\a{תה} \i{לי}:

„\a{ר}\i{אי}\a{ת}, \a{א}\x{ב}\a{ר}\x{ם}, \e{א}\x{ת} \a{מ}\x{ר}\e{אה} \a{ה}\x{ב}\e{ר}\x{י}\a{כה},\\

\e{א}\x{י}\x{ך} \o{קו}\e{ד}\x{ם} \a{ר}\x{ב}\a{צה} \a{ח}\a{ל}\a{קה} \e{ו}\o{דו}\e{מ}\e{מ}\x{ת},\\
\u{ו}\x{ל}\e{פ}\a{תע} \i{נ}\x{צ}\e{נ}\x{ץ} \e{ב}\a{פ}\e{ני}\a{ה} \a{ה}\e{ק}\e{מ}\x{ט},\\
\e{א}\x{י}\x{ך} \a{מ}\e{ש}\u{הו} \a{זע} \e{ו}\a{ר}\a{ח}\x{ש} \e{ב}\e{ח}\x{י}\a{ק}\x{ה},\\
\e{ו}\i{היא} \a{צ}\a{ח}\a{קה}, \a{צ}\a{ח}\a{קה}, \a{צ}\a{ח}\a{קה}?

\e{ל}\i{צי}\x{ם} \e{ש}\a{כ}\e{א}\e{לה}! \a{ד}\i{גי}\x{ם} \o{שו}\a{ב}\i{בי}\x{ם}!\\
\a{ר}\i{צי}\x{ם} \a{ש}\x{ם} \a{ב}\a{מ}\i{י}\x{ם}, \a{ר}\i{צי}\x{ם} \e{ו}\a{ש}\i{בי}\x{ם},\\
\e{ו}\e{זה} \e{מ}\a{ד}\x{ג}\e{ד}\x{ג} \a{לה}, \e{ו}\i{היא} \e{מ}\a{ח}\e{יי}\e{כ}\x{ת},\\

\e{ו}\e{ז}\u{הו} \a{ה}\a{ט}\a{ע}\x{ם} \i{ל}\x{ש}\a{מ}\x{ם}, \e{ש}\u{הוא} \a{ד}\x{ג}!”
